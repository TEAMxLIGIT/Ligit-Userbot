```console
 $ apt-get -y update
 $ apt-get -y install git gcc python3-pip -y
 $ git clone https://github.com/TeamxLigit/Ligit-Userbot
 $ cd Ligit-Userbot
 $ pip3 install -U -r requirements.txt
 $ cp sample.env .env
 $ nano .env
```
